Module 			::= { FunctionDef };

FunctionDef 	::= "fn", Identifier, "(", ")", Block ; 

Block 			::= "{", { Statement }, "}" ;

Statement 		::= "return" Literal ;

Identifier	  	::= Letter | "_" , { Letter | Digit | "_" } ;
Literal 		::= Digit, { Digit } ;

Letter 			::= "A".."Z" | "a".."z" ;
Digit			::= "0".."9" ;
